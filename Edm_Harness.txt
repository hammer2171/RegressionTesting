edmAriaHarness.ts — reusable ARIA snapshot helpers for EDMedm-aria.spec.ts — a test that uses your existing storageState and environmentAll snapshots are region‑scoped and tuned for EDM’s Oracle JET UI.

import { Page, Locator, expect } from '@playwright/test';

export class ViewpointHierarchyPage {
readonly region: Locator;
readonly newRequestButton: Locator;
readonly tree: Locator;

constructor(private page: Page) {
this.region = page.getByRole('application', { name: 'Viewpoint Hierarchy' });
this.newRequestButton = this.region.getByRole('button', { name: 'New Request' });
this.tree = this.region.getByRole('tree', { name: 'Hierarchy Tree' });
}

async waitForLoaded() {
await expect(this.region).toBeVisible();
}

async expandNode(name: string) {
await this.tree.getByRole('treeitem', { name }).click();
}

async openNewRequest() {
await this.newRequestButton.click();
}
}

// edm-aria.spec.ts
import { test, expect } from '@playwright/test';
import { EdmAriaHarness } from './edmAriaHarness';

test.use({
storageState: 'playwright/.auth/user.epm18_test.json'
});

test('EDM ARIA harness – request flow snapshots', async ({ page }) => {
const harness = new EdmAriaHarness(page);

// Login / landing
await page.goto('https://epm18-test-a706571.epm.us2.oraclecloud.com/epm/');
await harness.snapshotGlobalHeader();

// Open Views
await page.getByRole('img', { name: 'Views' }).click();
await harness.snapshotViewsList();

// Open specific viewpoint
await page.getByRole('link', { name: 'A_Entry_Entity' }).click();
await harness.snapshotViewpointHierarchy();

// Switch to Input_EPM_Entity_Base tab
await page.getByRole('tab', { name: 'Input_EPM_Entity_Base' }).click();
await harness.snapshotViewpointGrid();

// New Request flow
await page.getByRole('button', { name: 'New Request' }).click();
await expect(page.getByRole('grid', { name: 'List of Requests' })).toBeVisible();
await harness.snapshotRequestsRegion();

// Add Node
await page.getByRole('button', { name: 'Add Node' }).click();
await page.getByRole('menuitem', { name: 'Add New' }).click();
await expect(page.getByRole('row', { name: 'Request 35251 Show Item' })).toBeVisible();
await harness.snapshotAddNodeDialog();

// Ledger selection
await page.getByRole('textbox', { name: 'Ledger Number' }).click();
await expect(page.getByRole('row', { name: '3010LE (NOV Inc.)' })).toBeVisible();

// Node search + selection
await page.getByRole('textbox', { name: 'Search for a node' }).fill('0536');
await page.getByRole('textbox', { name: 'Search for a node' }).press('Enter');
await expect(page.getByRole('row', { name: '0536 (NOV Products Middle' })).toBeVisible();
await page.getByText('(NOV Products Middle East FZE)').click();
await page.getByRole('button', { name: 'OK' }).click();
await expect(page.getByRole('row', { name: 'Request 35251 Show Item' })).toBeVisible();

// Business Unit selection
await page
.locator(
'#oj-combobox-choice-50b08f12-f44f-4241-9e68-328f3b2a1622_e882c03c-72cc-4a88-be22-9c4d58207f07 > .oj-text-field-end > .oj-combobox-arrow'
)
.click();
await expect(page.getByRole('listbox', { name: 'Business Unit' })).toBeVisible();
await page.getByRole('option', { name: 'EE - Energy Equipment Admin' }).click();
await expect(page.getByRole('row', { name: 'Request 35251 Show Item' })).toBeVisible();

// Submit request
await page.getByRole('button', { name: 'Submit' }).click();
await expect(page.getByRole('tab', { name: 'Requests' })).toBeVisible();
await harness.snapshotRequestsRegion();

// Close and return home
await page.getByRole('button', { name: 'Close' }).click();
await harness.snapshotHomeAfterClose();

await page.getByRole('button', { name: 'Home' }).click();
});
