Command to run it - one at a time 

# 1) Run evidence test (trace + screenshots via evidence config)

$env:ENV_FILE=".env.epm18_test"
$env:PW_SLOWMO='0'
npm run test:e2e:evidence -- tests/oracle-epm-0057_import_all_maps.cleaned.spec.ts

# 2) Build markdown screenshot evidence doc

$env:ENV_FILE=".env.epm18_test"
$env:PW_SLOWMO='0'
npm run evidence:build


# 3) Extract images from trace.zip (replace <run-folder>)

$env:ENV_FILE=".env.epm18_test"
$env:PW_SLOWMO='0'
npm run evidence:extract-trace -- -TraceZip "C:\epm22_test\Mappings\ExportMapping\test-results\<run-folder>\trace.zip" -OutputDir ".\output\trace-images"

# 4) Build ordered markdown from extracted trace images

$env:ENV_FILE=".env.epm18_test"
$env:PW_SLOWMO='0'
npm run evidence:build-trace-doc -- -ImagesDir ".\output\trace-images" -OutputPath ".\output\trace-images.md"

# 5) Build pdf doc with one image per page (sheet)

$env:ENV_FILE=".env.epm18_test"
$env:PW_SLOWMO='0'
npm run evidence:build-trace-pdf -- -ImagesDir ".\output\trace-images" -OutputPath ".\output\trace-images.pdf"
